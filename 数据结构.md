  2. 标签表 tags

  {
    "id": "int",
    "code": "string",        // 编码，如 "ai", "intern"
    "name": "string",        // 名称，如 "人工智能", "实习"

  3. 职位信息表 jobs（核心表）

  {
    "id": "int",
    "title": "string",           // 职位名称
    "company_name": "string",    // 公司名称
    "industry_id": "int",        // 行业分类ID
    "industry_name": "string",   // 冗余存储便于查询

    // 投递信息
    "apply_email": "string",     // 投递邮箱
    "email_subject_template": "string", // 邮件标题模板
    "email_body_template": "string",    // 邮件正文模板

    // 岗位要求（JSON存储）
    "requirements": {
      "education": "本科及以上",     // 学历要求
      "experience": "应届生",         // 经验要求
      "location": "北京",             // 工作地点
      "skills": ["Python", "SQL"],   // 技能要求
      "salary": "15k-25k"            // 薪资范围
    },

    // 原始信息
    "source_url": "string",      // 来源链接
    "source_type": "公众号/官网/手动", // 来源类型
    "raw_content": "string",     // 原始文本备份
    "published_at": "datetime",  // 发布时间

    // 标签（多对多）
    "tags": ["ai", "intern", "remote"],

    // 状态
    "status": "draft/active/expired", // 草稿/有效/过期
    "created_at": "datetime",
    "updated_at": "datetime"
  }
